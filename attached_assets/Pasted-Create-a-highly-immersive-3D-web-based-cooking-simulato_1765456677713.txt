javascript// Storage utilities
const StorageKeys = {
  PLAYER: 'player-profile',
  INVENTORY: 'ingredient-inventory',
  RECIPES: 'recipe-progress',
  ACHIEVEMENTS: 'achievements-data',
  MEAL_PLAN: 'meal-planner',
  SETTINGS: 'game-settings',
  STATISTICS: 'player-statistics'
};

// Save player data
async function savePlayerData(playerData) {
  try {
    await window.storage.set(StorageKeys.PLAYER, JSON.stringify(playerData));
  } catch (error) {
    console.error('Failed to save player data:', error);
  }
}

// Load player data
async function loadPlayerData() {
  try {
    const result = await window.storage.get(StorageKeys.PLAYER);
    return result ? JSON.parse(result.value) : null;
  } catch (error) {
    console.error('Player data not found:', error);
    return null; // Return default data
  }
}

3D Performance Optimization
javascript// LOD (Level of Detail) system
- Close-up view: High-poly models (2000-5000 triangles)
- Medium view: Mid-poly models (500-1500 triangles)

// Texture optimization

- Mipmapping for distant objects
- Texture atlases to reduce draw calls
- Maximum texture size: 2048x2048

// Rendering optimization
- Frustum culling (only render visible objects)
- Occlusion culling (don't render hidden objects)
- Instanced rendering for repeated objects
- Object pooling for particles
- Limit active particle systems (max 5 concurrent)

// Frame rate targeting
- Desktop: 60 FPS
- Mobile: 30 FPS
- Adaptive quality: reduce effects if FPS drops

// Memory management
- Dispose geometries and materials when unused
- Unload unused 3D models
- Limit simultaneous loaded recipes (max 3)

javascript// Chopping Game Component
function ChoppingGame({ ingredient, onComplete }) {
  const [cuts, setCuts] = useState(0);
  const [accuracy, setAccuracy] = useState(100);
  const targetCuts = 12;
  
  const handleChop = (position) => {
    // Calculate cut accuracy based on position
    const accuracyScore = calculateCutAccuracy(position);
    
    // Update 3D visual - slice ingredient
    sliceIngredient3D(ingredient, position);
    
    // Play sound effect
    playSound('chop', {pitch: 1 + (cuts * 0.1)});
    
    // Particle effect
    createChopParticles(position);
    
    setCuts(prev => prev + 1);
    setAccuracy(prev => (prev + accuracyScore) / 2);
    
    if (cuts + 1 >= targetCuts) {
      const finalScore = calculateScore(cuts, accuracy, timeElapsed);
      onComplete({score: finalScore, stars: getStars(finalScore)});
    }
  };
  
  return (
    <Canvas>
      <CuttingBoard3D />
      <Ingredient3D model={ingredient} cuts={cuts} />
      <Knife3D onChop={handleChop} />
      <ChopGuideOverlay />
    </Canvas>
  );
}

function App() {
  const [gameState, setGameState] = useState('loading'); // loading, home, cookbook, cooking
  const [playerData, setPlayerData] = useState(null);
  const [currentRecipe, setCurrentRecipe] = useState(null);
  
  useEffect(() => {

    }

  if (gameState === 'loading') {
    return <LoadingScreen />;
  }
  
  return (
    <div className="app-container">
      {gameState === 'home' && (
        <HomeScreen 
          playerData={playerData}
          onStartCooking={() => setGameState('cookbook')}
        />
      )}
      
      {gameState === 'cookbook' && (
        <RecipeBook 
          playerData={playerData}
          onSelectRecipe={(recipe) => {
            setCurrentRecipe(recipe);
            setGameState('cooking');
          }}
        />
      )}
      
      {gameState === 'cooking' && (
        <CookingScreen 
          recipe={currentRecipe}
          playerData={playerData}
          onComplete={(results) => {
            updatePlayerProgress(results);
            setGameState('home');
          }}
        />
      )}
    </div>
  );
}

export default App;
3D Kitchen Scene
javascriptimport React, { useRef } from 'react';
import { useFrame } from '@react-three/fiber';
import Stove from './Stove';
import Countertop from './Countertop';
import Refrigerator from './Refrigerator';

function KitchenScene({ activeStation }) {
  const groupRef = useRef();
  
  useFrame((state) => {
    // Subtle ambient animation
    if (groupRef.current) {
      groupRef.current.rotation.y = Math.sin(state.clock.elapsedTime * 0.1) * 0.02;
    }
  });
  
  return (
    <group ref={groupRef}>
      {/* Lighting */}
      <ambientLight intensity={0.4} />
      <directionalLight position={[5, 10, 5]} intensity={0.8} castShadow />
      <pointLight position={[0, 3, 0]} intensity={0.5} color="#FFF8E7" />
      
      {/* Kitchen Elements */}
      <Countertop position={[0, 0, 0]} />
      <Stove position={[2, 1, 0]} active={activeStation === 'stove'} />
      <Refrigerator position={[-3, 0, -2]} />
      
      {/* Floor */}
      <mesh rotation={[-Math.PI / 2, 0, 0]} position={[0, -0.01, 0]} receiveShadow>
        <planeGeometry args={[20, 20]} />
        <meshStandardMaterial color="#E8D5B5" />
      </mesh>
      
      {/* Walls */}
      <mesh position={[0, 3, -5]}>
        <planeGeometry args={[20, 6]} />
        <meshStandardMaterial color="#F5F5DC" />
      </mesh>
    </group>
  );
}

export default KitchenScene;
